# exp 和 log 函数

  • exp(n) 值为 ene^nen
  • log() 以 eee 为底（即 ln⁡\lnln）
  • log10() 以 101010 为底
#include <iostream>
#include <cmath>
using namespace std;

int main() {
    double a = 9, b = 10;
    cout << log(a) << endl;       // ln9
    cout << log(exp(a)) << endl;  // 9
    cout << log10(b) << endl;     // 1
    return 0;
}

自定义以 m 为底，求 log n 的值：

double a = log(n) / log(m);

示例：以 2 为底的对数函数

# 并查集

P3367

#include <bits/stdc++.h>
using namespace std;

int n, m;
int r[10001];
int fa[10001];

void init(int n) {
    for (int i = 1; i <= n; ++i) {
        fa[i] = i;
        r[i] = 1;
    }
}

// 查找根节点（路径压缩）
int find(int x) {
    return x == fa[x] ? x : (fa[x] = find(fa[x]));
}

// 合并（按秩合并）
inline void merge(int i, int j) {
    int x = find(i), y = find(j);
    if (r[x] <= r[y])
        fa[x] = y;
    else
        fa[y] = x;
    if (r[x] == r[y] && x != y)
        r[y]++;  // 如果深度相同且根节点不同，则新的根节点的深度+1
}

int main() {
    cin >> n >> m;
    init(n);
    for (int i = 0; i < m; i++) {
        int t, x, y;
        cin >> t >> x >> y;
        if (t == 1) merge(x, y);
        else {
            if (find(x) == find(y)) cout << "Y" << endl;
            else cout << "N" << endl;
        }
    }
    return 0;
}

# Manacher（字符串中的最长回文子串）

P3805

#include <bits/stdc++.h>
using namespace std;

int p[33000001];

int manacher(string s) {
    string st = "$#";
    for (int i = 0; i < s.size(); i++) {
        st += s[i];
        st += "#";
    }
    int mid = 0, r = 0, ans = 0;
    for (int i = 1; i < st.size(); i++) {
        p[i] = r > i ? min(p[mid * 2 - i], r - i) : 1;
        while (st[i - p[i]] == st[i + p[i]]) p[i]++;
        if (p[i] > ans) ans = p[i];
        if (i + p[i] > r) {
            r = i + p[i];
            mid = i;
        }
    }
    return ans - 1;
}

int main() {
    string s;
    cin >> s;
    cout << manacher(s);
    return 0;
}

# KMP

P3375

#include <bits/stdc++.h>
using namespace std;

int Next[1000002];
string s1, s2;

void get_Next(string s) {
    s += "#";
    Next[0] = -1;
    int k = -1, i = 0, l = s.size();
    while (i < l) {
        if (s[i] == s[k] || k == -1) {
            i++;
            k++;
            Next[i] = k;
        } else {
            k = Next[k];
        }
    }
}

void kmp(string s1, string s2) {
    int l1 = s1.size(), l2 = s2.size();
    int i = 0, j = 0;
    while (j < l1) {
        if (s2[i] == s1[j]) {
            i++;
            j++;
            if (i == l2) {           // 完全匹配
                cout << j - l2 + 1 << endl;
                i = Next[i - 1] + 1; // 继续匹配
            }
        } else {
            if (i == 0) j++;              // 如果在首位不匹配
            else i = Next[i - 1] + 1;     // 跳转
        }
    }
}

int main() {
    cin >> s1 >> s2;
    get_Next(s2);
    kmp(s1, s2);
    for (int i = 1; i <= s2.size(); i++) cout << Next[i] << " ";
    return 0;
}

# 拓扑排序

P1113

#include <bits/stdc++.h>
using namespace std;

vector<int> edge[10001];  // 存储当前节点指向结点
int tim[10001], ind[10001] = {0}, f[10001] = {0};  // 时间，入度，答案
int n;
queue<int> q;

int main() {
    cin >> n;
    for (int i = 1; i <= n; i++) {
        int w;
        cin >> w;
        cin >> tim[i];
        int p;
        cin >> p;
        while (p) {
            edge[p].push_back(i);
            ind[i]++;
            cin >> p;
        }
    }

    // 寻找最开始入度就是0的点，入列
    for (int i = 1; i <= n; i++) {
        if (!ind[i]) {
            q.push(i);
            f[i] = tim[i];
        }
    }

    // 找每一个队列中的点的相邻节点
    while (!q.empty()) {
        int temp = q.front();
        q.pop();
        int l = edge[temp].size();
        for (int i = 0; i < l; i++) {
            ind[edge[temp][i]]--;
            if (!ind[edge[temp][i]]) q.push(edge[temp][i]);
            f[edge[temp][i]] = max(f[edge[temp][i]], f[temp] + tim[edge[temp][i]]);
        }
    }

    int ans = 0;
    for (int i = 1; i <= n; i++) {
        ans = max(ans, f[i]);
    }
    cout << ans;
    return 0;
}

# 单点修改，区间查询（维护原数组）

P3374

#include <bits/stdc++.h>
#define lowbit(x) (x & (-x))
using namespace std;

const int N = 2000006;
int c[N] = {0};
int n, m;

// 单点修改
void change_onepoint(int ind, int k) {
    for (int i = ind; i <= n; i += lowbit(i)) c[i] += k;
}

// 区间查询
void search(int l, int r) {
    long long ans = 0;
    for (int i = r; i; i -= lowbit(i)) ans += c[i];
    for (int i = l - 1; i; i -= lowbit(i)) ans -= c[i];
    cout << ans << endl;
}

int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; i++) {
        int t;
        cin >> t;
        change_onepoint(i, t);
    }
    while (m--) {
        int t, x, y;
        cin >> t >> x >> y;
        if (t == 1) change_onepoint(x, y);
        else search(x, y);
    }
    return 0;
}

# 区间修改，单点查询（维护差分数组）

P3368（区间用单点，单点用区间）

#include <bits/stdc++.h>
#define lowbit(x) (x & (-x))
using namespace std;

const int N = 1000006;
int c[N] = {0};       // 树状数组
int a[N] = {0};       // 原数组
int chafen[N] = {0};  // 差分数组
int n, m;

// 对差分数组的单点修改
void change_onepoint(int ind, int k) {
    for (int i = ind; i <= n; i += lowbit(i)) c[i] += k;
}

// 对差分数组的区间查询，其实就是求原数组的单点值
void search(int r) {
    long long ans = 0;
    for (int i = r; i; i -= lowbit(i)) ans += c[i];
    cout << ans << endl;
}

int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        chafen[i] = a[i] - a[i - 1];
        change_onepoint(i, chafen[i]);
    }
    while (m--) {
        int t;
        cin >> t;
        if (t == 1) {
            int x, y, k;
            cin >> x >> y >> k;
            change_onepoint(x, k);
            change_onepoint(y + 1, -k);
        } else {
            int k;
            cin >> k;
            search(k);
        }
    }
    return 0;
}

# 区间修改，区间查询（超级树状数组）

P3372（本来是线段树的板子，用超级树状数组做）

对于 a 的树状数组（差分）tree，建立一个新的树状数组 tree1 使得：

tree1[i] = tree[i] * (i - 1)

之后，x 到 y 的区间和即为：

(y * getsum(tree, y) - (x-1) * getsum(tree, x-1)) - (getsum(tree1, y) - getsum(tree1, x-1))

Tips: 因为求区间和满足区间加法，所以 Sum(L,R) = Sum(1,R) - Sum(1,L-1)，所以有上述公式。

# 最小生成树（Prim）

P3366（可以囊括两节点之间有多条路径的情况）P1194 P1396

#include <bits/stdc++.h>
using namespace std;

int n, m, a, b, v, cnt = 0;

struct Edge {
    int fr, to, l;
    explicit Edge(int aa = 0, int bb = 0, int vv = 0) {
        fr = aa; to = bb; l = vv;
    }
    bool operator<(const Edge& o) const {
        return this->l > o.l;
    }
};

vector<Edge> graph[100010];
bool flag[100010];  // flag[i]==true means node i is in tree
Edge tree[100010];

void Prim() {
    priority_queue<Edge> pq;
    flag[1] = true;
    for (Edge e : graph[1]) {
        pq.push(e);
    }
    while (cnt < n - 1 && !pq.empty()) {
        Edge top = pq.top();
        pq.pop();
        if (flag[top.to]) continue;  // 当前节点已经使用过
        flag[top.to] = true;
        tree[cnt++] = top;
        for (Edge e : graph[top.to]) {
            if (!flag[e.to]) {
                pq.push(e);
            }
        }
    }
}

int main() {
    scanf("%d %d", &n, &m);
    for (int i = 0; i < m; ++i) {
        cin >> a >> b >> v;
        Edge e1(a, b, v), e2(b, a, v);
        graph[a].push_back(e1);
        graph[b].push_back(e2);
    }
    Prim();
    if (cnt < n - 1) {
        printf("orz");
        return 0;
    }
    int ans = 0;
    for (int i = 0; i < n - 1; ++i) {
        Edge e = tree[i];
        ans += e.l;
    }
    printf("%d", ans);
    return 0;
}

# 最短路径（Dijkstra）

P1359（用最小生成树改造的模板）

#include <bits/stdc++.h>
using namespace std;

int n, v;

struct Edge {
    int fr, to, l;
    explicit Edge(int aa = 0, int bb = 0, int vv = 0) {
        fr = aa; to = bb; l = vv;
    }
    bool operator<(const Edge& o) const {
        return this->l > o.l;
    }
};

vector<Edge> graph[100010];
bool flag[210] = {0};
int dis[210];
const int INF = 1e9;

void Dijkstra() {
    flag[1] = 1;
    int u = 1;
    int mi = INF, s = -1;
    for (Edge e : graph[1]) {
        dis[e.to] = e.l;
        if (dis[e.to] < mi) {
            mi = dis[e.to];
            s = e.to;
        }
    }
    u = s;
    while (!flag[n]) {
        for (Edge e : graph[u]) {
            dis[e.to] = min(dis[e.to], dis[u] + e.l);
        }
        mi = INF;
        for (int i = 1; i <= n; i++) {
            if (!flag[i] && dis[i] < mi) {
                mi = dis[i];
                s = i;
            }
        }
        if (s == -1) {
            cout << "NO WAY";
            return;  // 所有未访问节点都不可达
        }
        u = s;
        flag[u] = 1;
    }
}

int main() {
    cin >> n;
    for (int i = 1; i < n; ++i) {
        for (int j = i + 1; j <= n; j++) {
            cin >> v;
            Edge e(i, j, v);
            graph[i].push_back(e);
        }
    }
    fill(dis, dis + n + 1, 1000001);
    Dijkstra();
    cout << dis[n];
    return 0;
}

# GCD（最大公约数）

int gcd(int x, int y) {
    int temp;
    while (y != 0) {
        temp = x % y;
        x = y;
        y = temp;
    }
    return x;
}

# DFS 搜索（连通块计数）

#include <bits/stdc++.h>
using namespace std;

int dx[8] = {0, 0, -1, 1, -1, 1, 1, -1};
int dy[8] = {1, -1, 0, 0, 1, 1, -1, -1};
int r, c;
char maps[110][110];

void dfs(int x, int y) {
    if (x < 0 || x > r + 1 || y < 0 || y > c + 1 || maps[x][y] != 'W') return;
    maps[x][y] = '*';
    for (int i = 0; i < 8; i++) {
        dfs(x + dx[i], y + dy[i]);
    }
}

int main() {
    cin >> r >> c;
    for (int i = 1; i <= r; i++) {
        for (int j = 1; j <= c; j++) {
            cin >> maps[i][j];
        }
    }
    int ans = 0;
    for (int i = 1; i <= r; i++) {
        for (int j = 1; j <= c; j++) {
            if (maps[i][j] == 'W') {
                ans++;
                dfs(i, j);
            }
        }
    }
    cout << ans;
    return 0;
}

#include <bits/stdc++.h>
using namespace std;

int f[20001];
int w[31];

int main() {
    int V, n;
    cin >> V >> n;
    f[0] = 0;
    for (int i = 1; i <= n; i++) cin >> w[i];
    for (int i = 1; i <= n; i++) {
        for (int j = V; j >= w[i]; j--) {
            if (f[j] < f[j - w[i]] + w[i]) f[j] = f[j - w[i]] + w[i];
        }
    }
    cout << V - f[V];
    return 0;
}
